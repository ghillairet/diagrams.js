Getting Started

Diagram

Create a diagram. 

                    
var diagram = new Diagram.Diagram('canvas', 800, 600);
diagram.render();

Shapes

Shapes are defined by extending the class Diagram.Shape. 

                    
var SomeShape = Diagram.Shape.extend({
    figure: {
        type: 'circle',
        r: 30,
        fill: 'blue',
        stroke: 'rgb(120, 120, 200)',
        'stroke-width': 2
    }
});
                    
                

                    
var shape = new SomeShape({diagram: diagram, x: 10, y: 10 });
shape.render();

Connections

Connections are defined by extending the class Diagram.Shape. 

                    
var BasicConnection = Diagram.Connection.extend({
    stroke: 'red',
    'stroke-width': 2,
    label: [
        { position: 'end', text: 'label' }
    ],
    end: {
        type: "none"
    },
    start: {
        type: "none"
    }
});

jsfiddle full screen 
                    
var connection = new BasicConnection({diagram: diagram});
connection.connect(sourceShape, targetShape);
connection.render();
